    <script>

        document.addEventListener('DOMContentLoaded', () => {

            const generateButton = document.getElementById('generateButton');

            const designKeywordInput = document.getElementById('designKeywordInput');

            const designOutput = document.getElementById('designOutput');

            const loadingIndicator = document.getElementById('loadingIndicator');

 

            // API key untuk Gemini

            const apiKey = "";

            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-05-20:generateContent?key=${apiKey}`;

 

            // Fungsi untuk memanggil Gemini API

            async function generateDesignIdea(prompt) {

                const systemPrompt = "Bertindak sebagai perancang stiker sablon yang kreatif. Berikan 3-5 ide desain yang unik, menarik, dan singkat berdasarkan kata kunci yang diberikan. Format jawabannya sebagai daftar bernomor atau poin.";

                const userQuery = `Buat ide desain sablon untuk tema: ${prompt}`;

 

                const payload = {

                    contents: [{ parts: [{ text: userQuery }] }],

                    systemInstruction: {

                        parts: [{ text: systemPrompt }]

                    },

                };

 

                // Lakukan retry dengan exponential backoff

                let response;

                for (let i = 0; i < 3; i++) { // Coba maksimal 3 kali

                    try {

                        response = await fetch(apiUrl, {

                            method: 'POST',

                            headers: { 'Content-Type': 'application/json' },

                            body: JSON.stringify(payload)

                        });

                        if (response.ok) {

                            break; // Berhasil, keluar dari loop

                        }

                    } catch (e) {

                        console.error(`Fetch attempt ${i + 1} failed:`, e);

                    }

                    if (i < 2) {

                        await new Promise(resolve => setTimeout(resolve, Math.pow(2, i) * 1000));

                    }

                }

 

                if (!response || !response.ok) {

                    throw new Error("Failed to fetch from API after multiple retries.");

                }

 

                const result = await response.json();

                const candidate = result.candidates?.[0];

 

                if (candidate && candidate.content?.parts?.[0]?.text) {

                    return candidate.content.parts[0].text;

                } else {

                    return "Maaf, tidak dapat menghasilkan ide saat ini. Silakan coba lagi.";

                }

            }

 

            // Tambahkan event listener untuk tombol

            generateButton.addEventListener('click', async () => {

                const keywords = designKeywordInput.value.trim();

                if (keywords === "") {

                    designOutput.textContent = "Silakan masukkan kata kunci terlebih dahulu.";

                    return;

                }

 

                generateButton.disabled = true;

                loadingIndicator.classList.remove('hidden');

                designOutput.textContent = ''; // Hapus konten sebelumnya

 

                try {

                    const ideas = await generateDesignIdea(keywords);

                    designOutput.textContent = ideas;

                } catch (error) {

                    console.error("Error generating design ideas:", error);

                    designOutput.textContent = "Terjadi kesalahan. Silakan coba lagi nanti.";

                } finally {

                    generateButton.disabled = false;

                    loadingIndicator.classList.add('hidden');

                }

            });

        });

    </script>
